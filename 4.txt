 SELECT 'New Customers' AS CUSTOMER_TYPE,  SUM(A.TOTAL_PRICE - A.TOTAL_TAX) TOTAL_REVENUE FROM ORDERS A WHERE A.CUSTOMER_ID IN (SELECT X.CUSTOMER_ID FROM (SELECT CUSTOMER_ID, COUNT(*) FROM ORDERS GROUP BY CUSTOMER_ID HAVING COUNT(*) = 1
 ) X)
UNION ALL
  SELECT 'Returning Customers' AS CUSTOMER_TYPE, SUM(A.TOTAL_PRICE - A.TOTAL_TAX) TOTAL_REVENUE FROM ORDERS A WHERE A.CUSTOMER_ID IN (SELECT X.CUSTOMER_ID FROM (SELECT CUSTOMER_ID, COUNT(*) FROM ORDERS GROUP BY CUSTOMER_ID HAVING COUNT(*) > 1
 ) X)
